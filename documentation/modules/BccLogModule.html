import {CommonModule} from '@angular/common';
import {HttpClientModule} from '@angular/common/http';
import {ModuleWithProviders, NgModule} from '@angular/core';

import {BCCLoggerService} from './logger.service';
import {LoggerConfig} from './logger.config';
import {CustomBCCLoggerService} from './custom-logger.service';
import {BCCLoggerHttpService} from './http.service';

export * from './logger.service';

export * from './logger.config';

export * from './custom-logger.service';

export * from './logger-monitor';

export * from './http.service';

export * from './utils/logger.utils';
export * from './types/logger-level.enum';
export * from './types/bcc-log.interface';

@NgModule({
  imports: [CommonModule, HttpClientModule],
  providers: [BCCLoggerService, BCCLoggerHttpService, CustomBCCLoggerService],
})
export class BccLogModule {
  static forRoot(config: LoggerConfig | null | undefined): ModuleWithProviders {
    return {
      ngModule: BccLogModule,
      providers: [
        {provide: LoggerConfig, useValue: config || {}},
        BCCLoggerService,
        BCCLoggerHttpService,
        CustomBCCLoggerService,
      ],
    };
  }
  static forChild(): ModuleWithProviders {
    return {
      ngModule: BccLogModule,
      providers: [BCCLoggerService, BCCLoggerHttpService, CustomBCCLoggerService],
    };
  }
}
